1.) create a secret for your repo:

echo 'apiVersion: v1
kind: Secret
metadata:
  name: homelab-repo
  labels:
    chart: "{{ .Chart.Name }}-{{ .Chart.Version | replace "+" "_" }}"
  annotations:
    sealedsecrets.bitnami.com/namespace-wide: "true"
  namespace: argocd
stringData:
  {{- range $key, $val := .Values.reposecrets }}
    {{ $key }}: {{ $val | quote }}
  {{- end }}
' > argocd-repository.yaml

2.) encrypt is using sealed secrets and apply it to the cluster

cat argocd-repository.yaml | kubeseal | kubectl apply -f -
